What do I want out of classes in quartz?:

I want to be able to have an object like structure that can
- have named fields that are readable, but non-writable by default
- be able to implement interfaces to allow polymorphism
- have functions be public by default

Example:

using std.Math;
using std.String;
using std.printfln;

public interface Stringable {
    String toString();
}

public class Point2D : Stringable {

    Double x;
    Double y;

    Point2D(Double x, Double y){
        this.x = x;
        this.y = y;
    }

    Point2D(){
        this.x = 0;
        this.y = 0;
    }

    String toString(){
        return String.format("Point2D:{x: $1, y: $2}", [this.x, this.y]);
    }

}

Double distance(Point2D a, Point2D b){
    return Math.sqrt(Math.square(a.x - b.x) + Math.square(a.y - b.y));
}

void main(){

    var pointA = new Point2D(1.0, 1.0);
    var pointB = new Point2D(2.0, 2.0);

    println("Point A is: " + pointA.toString());
    println("Point B is: " + pointB.toString());

    printfln(
        "The distance between Point A and B is: $1",
        [distance(pointA, pointB)]
    );

}
