<code> ::= (<variable-declaration> | <variable-initialization> ) <whitespace> 

<global_code> ::= (<library-import> | <struct-define> | <typecast-function> ) <whitespace> <global_code> | <end-of-file>

<library-import> ::= "using" <library-path> ";" 

<struct-define> ::= "struct" <identifier> <whitespace> "{" {<datatype> <identifier> ";" <whitespace> } "}" 

<typecast-function> ::= "typecast " <datatype> "(" <datatype> ")" <whitespace> "{" <code> "}"

<datatype> ::= <type>["*" | "[]" ]

<type> ::= <float> | <integer> | "bool" | <struct>

<float> ::= "double"

<integer> ::= "int" | "uint" | "char"

<struct> ::= "string"

<whitespace> ::= { " " | "\n" | "\t" | "\r\n" }
